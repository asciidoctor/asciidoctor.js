= Block Macro Extension Example

Purpose::
Generate lorem ipsum text.

== sample-lorem-doc.adoc

[source,adoc]
----
lorem::sentences[size=2]
----

== LoremBlockMacro

.lorem-block-macro.js
[source,js]
----
include::example$extensions/lorem-block-macro.js[]
----

== Usage

[source,js]
----
import { Extensions, convert } from '@asciidoctor/core'
import registerLoremBlockMacro from './lorem-block-macro.js'

const registry = Extensions.create()
registerLoremBlockMacro(registry)

const html = await convert('lorem::sentences[size=2]', { extension_registry: registry })
console.log(html)
// <div class="paragraph">
// <p>Lorem ipsum dolor sit amet...</p>
// </div>
----
